## 0. Decision & Constraints (read before any code)

**Approach: Option A — static JSON export → Next.js static site on Vercel.**
A new pipeline step dumps the existing `data/sample/` artifacts to JSON. A Next.js app
(`output: 'export'`) reads them as static files. The transport-cost slider re-ranks mandis
**client-side in TypeScript**, porting `src/mandipulse/recommend/engine.py` exactly. No backend,
no second server, numbers identical to Streamlit.

**Hard constraints (violating any = wrong):**
- `next.config.ts` must use `output: 'export'` (static). **No** `output: 'server'`, no SSR data fetching.
- All data is `import`-ed / `fetch`-ed from static JSON in `public/data/`. No runtime API calls.
- The TS ranking must match `engine.py` to within 0.01 INR/qtl at the default location (parity test, N-10).
- MVP data scope unchanged: 1 crop (onion), 1 state (maharashtra), 7-day horizon, 15 mandis.
- No generated-code attribution trailers in commits. Commit/push only when the human asks.
- Do not invent config values — read them from `configs/recommendation.yaml`.

## 4. Phase N-02 — `scripts/build_web_export.py`

Mirror the structure of `build_recommendations_7d.py` (sys.path insert, `load_yaml_config`).
**Read inputs from `data/sample/`** (the committed bundle), falling back to full `artifacts/` paths
via `paths.py` if present — same spirit as `data_access.py`. Reuse `score_recommendations` so the
default ranking is byte-identical logic to Streamlit.

```python
from __future__ import annotations
import json, sys
from pathlib import Path
import pandas as pd

sys.path.insert(0, str(Path(__file__).resolve().parents[1] / "src"))
from mandipulse.config import load_yaml_config  # noqa: E402
from mandipulse.recommend.engine import score_recommendations  # noqa: E402

SAMPLE = Path("data/sample")
MANDIS_CSV = Path("data/external/mvp_mandis.csv")
OUT = Path("web/public/data")
DEFAULT_LAT, DEFAULT_LON = 19.99750, 73.78981

def main() -> int:
    cfg = load_yaml_config("configs/recommendation.yaml")
    tc, rk, rt = cfg["transport_cost"], cfg["ranking"], cfg["risk_thresholds"]
    OUT.mkdir(parents=True, exist_ok=True)

    forecasts = pd.read_csv(SAMPLE / "forecast_outputs_7d.csv")
    prices = pd.read_csv(SAMPLE / "clean_mandi_prices.csv", parse_dates=["date"])
    backtest = pd.read_csv(SAMPLE / "recommendation_backtest_7d.csv")
    mandis = pd.read_csv(MANDIS_CSV)
    mandis["district_name"] = mandis["district_name"]  # already present

    # latest forecast per market_id
    latest = forecasts.sort_values("as_of_date").groupby("market_id").tail(1)

    # --- meta.json ---
    meta = {
        "as_of_date": str(latest["as_of_date"].max()),
        "crop": "onion", "state": "maharashtra",
        "model_version": str(latest["model_name"].iloc[0]),
        "confidence_level": float(latest["confidence_level"].iloc[0]),
        "empirical_coverage": 0.8671,
        "default_farmer": {"latitude": DEFAULT_LAT, "longitude": DEFAULT_LON},
        "ranking": {
            "cost_per_km_per_quintal": float(tc["cost_per_km_per_quintal"]),
            "road_distance_factor": float(tc["road_distance_factor"]),
            "max_transport_radius_km": float(tc["max_transport_radius_km"]),
            "uncertainty_penalty_weight": float(rk["uncertainty_penalty_weight"]),
            "low_max_interval_pct": float(rt["low_max_interval_pct"]) / 100,   # 0.10
            "high_min_interval_pct": float(rt["high_min_interval_pct"]) / 100, # 0.25
            "cost_variation_pct": float(cfg["sensitivity"]["transport_cost_variation_pct"]),
        },
    }
    _dump("meta.json", meta)

    # --- mandis.json ---
    mcols = ["market_id","market_name","district_name","latitude","longitude","active_days"]
    _dump("mandis.json", mandis[mcols].to_dict("records"))

    # --- forecasts.json (compute risk_level via the SAME engine, not stored in CSV) ---
    ranked_default = score_recommendations(
        forecasts=latest, mandis=mandis,
        farmer_latitude=DEFAULT_LAT, farmer_longitude=DEFAULT_LON,
        cost_per_km_per_quintal=tc["cost_per_km_per_quintal"],
        road_distance_factor=tc["road_distance_factor"],
        uncertainty_penalty_weight=rk["uncertainty_penalty_weight"],
        low_max_interval_pct=rt["low_max_interval_pct"]/100,
        high_min_interval_pct=rt["high_min_interval_pct"]/100,
        candidate_state="maharashtra",
    )
    fcols = ["market_id","mandi_id","mandi","as_of_date","forecast_price_inr_qtl",
             "lower_bound_inr_qtl","upper_bound_inr_qtl"]
    flat = latest.merge(
        ranked_default[["market_id","risk_level"]], on="market_id", how="left")
    _dump("forecasts.json", flat[fcols + ["risk_level"]].to_dict("records"))

    # --- recommendations.json (default-location ranking, full ranked output) ---
    rcols = ["rank","market_id","mandi","district_name","forecast_price_inr_qtl",
             "lower_bound_inr_qtl","upper_bound_inr_qtl","estimated_transport_cost_inr_qtl",
             "expected_net_price_inr_qtl","uncertainty_penalty_inr_qtl",
             "risk_adjusted_score","risk_level","air_distance_km","road_distance_km","reason"]
    _dump("recommendations.json", ranked_default[rcols].round(4).to_dict("records"))

    # --- price_history.json (last 180 days, keep is_imputed) ---
    cutoff = prices["date"].max() - pd.Timedelta(days=180)
    recent = prices[prices["date"] >= cutoff].copy()
    recent["date"] = recent["date"].dt.strftime("%Y-%m-%d")
    pcols = ["market_id","market_name","date","modal_price_inr_qtl","is_imputed"]
    _dump("price_history.json", recent[pcols].to_dict("records"))

    # --- backtest.json (summary) ---
    bt = {
        "mean_regret_at_1": round(float(backtest["regret_at_1"].mean()), 2),
        "nearest_mandi_baseline_regret": round(float(backtest["nearest_mandi_regret"].mean()), 2),
        "pct_beats_nearest": round(float(
            (backtest["regret_at_1"] < backtest["nearest_mandi_regret"]).mean() * 100), 1),
        "n_dates_evaluated": int(len(backtest)),
        "test_window_start": str(backtest["as_of_date"].min()),
        "test_window_end": str(backtest["as_of_date"].max()),
    }
    _dump("backtest.json", bt)

    # --- honest_results.json (committed report numbers, hardcoded) ---
    _dump("honest_results.json", [
        {"model": "moving_average_7d", "test_mae": 139.57, "ships": True},
        {"model": "ridge", "test_mae": 224.43, "ships": False},
        {"model": "lightgbm", "test_mae": 188.2, "ships": False},
        {"model": "lightgbm_residual", "test_mae": 195.63, "ships": False},
    ])
    return 0

def _dump(name: str, obj) -> None:
    (OUT / name).write_text(json.dumps(obj, indent=2), encoding="utf-8")
    print(f"wrote web/public/data/{name}")

if __name__ == "__main__":
    raise SystemExit(main())
```

## 7. Phase N-05 — `lib/transport.ts` (exact port of `engine.py`)

```typescript
// Port of src/mandipulse/recommend/engine.py — keep in sync. atan2 (NOT asin), road = air × factor.
const RADIUS_KM = 6371.0

export function haversineKm(lat1: number, lon1: number, lat2: number, lon2: number): number {
  const phi1 = (lat1 * Math.PI) / 180, phi2 = (lat2 * Math.PI) / 180
  const dPhi = ((lat2 - lat1) * Math.PI) / 180, dLambda = ((lon2 - lon1) * Math.PI) / 180
  const a = Math.sin(dPhi / 2) ** 2 + Math.cos(phi1) * Math.cos(phi2) * Math.sin(dLambda / 2) ** 2
  return 2 * RADIUS_KM * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a))
}

export function riskLevel(relWidth: number, lowMax: number, highMin: number): string {
  if (relWidth <= lowMax) return "low"
  if (relWidth < highMin) return "medium"
  return "high"
}

export interface RankParams {
  costPerKm: number; roadDistanceFactor: number; uncertaintyPenaltyWeight: number
  lowMaxPct: number; highMinPct: number   // already as ratios: 0.10, 0.25
}

export function rankMandis(forecasts, mandis, farmerLat, farmerLon, p: RankParams) {
  return forecasts
    .map(f => {
      const m = mandis.find(x => x.market_id === f.market_id)!
      const airKm = haversineKm(farmerLat, farmerLon, m.latitude, m.longitude)
      const roadKm = airKm * p.roadDistanceFactor
      const transport = roadKm * p.costPerKm
      const netPrice = f.forecast_price_inr_qtl - transport
      const width = f.upper_bound_inr_qtl - f.lower_bound_inr_qtl
      const penalty = width * p.uncertaintyPenaltyWeight
      const riskAdjusted = netPrice - penalty
      const relWidth = width / Math.max(f.forecast_price_inr_qtl, 1.0)
      return { ...f, market_name: m.market_name, district_name: m.district_name,
               airKm, roadKm, transport, netPrice, penalty, riskAdjusted,
               risk: riskLevel(relWidth, p.lowMaxPct, p.highMinPct) }
    })
    .sort((a, b) => b.riskAdjusted - a.riskAdjusted || b.netPrice - a.netPrice)
    .map((r, i) => ({ ...r, rank: i + 1 }))
}
```
Defaults from `meta.json.ranking`: `costPerKm=4.0`, `roadDistanceFactor=1.3`,
`uncertaintyPenaltyWeight=0.3`, `lowMaxPct=0.10`, `highMinPct=0.25`.
